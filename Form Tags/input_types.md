<!--

1. text
- Default input type.
- Used for single-line text.

Example:
<input type="text">

2. password
- Hides the entered characters.

Example:
<input type="password">

3. date
- Used to select a date.

Example:
<input type="date">

4. checkbox
- Allows selecting multiple options.

Example:
<input type="checkbox">

5. color
- Opens a color picker.

Example:
<input type="color">

6. button
- Creates a clickable button.

Example:
<input type="button" value="Click">

7. datetime-local
- Selects both date and time.

Example:
<input type="datetime-local">

8. email
- Accepts only valid email addresses.

Example:
<input type="email">

9. hidden
- Stores hidden data.
- It is sent to the server but is not visible to users.

Example:
<input type="hidden" name="userid" value="101">

10. image
- Displays an image as a submit button.

Example:
<input type="image" src="submit.png" alt="Submit">

11. file
- Used to upload files.

Example:
<input type="file">

12. month
- Selects only month and year.

Example:
<input type="month">

13. number
- Accepts numeric values only.
- Can be restricted using min and max.

Example:
<input type="number" name="quantity" min="0" max="100">

14. radio
- Allows selecting only one option from a group.

Example:
<input type="radio" name="gender" value="Male">
<input type="radio" name="gender" value="Female">

15. range
- Creates a slider.

Example:
<input type="range" name="range" min="0" max="100">

16. tel
- Used for telephone numbers.

Example:
<input type="tel">

17. reset
- Resets all form fields to their default values.

Example:
<input type="reset">

18. search
- Used for search input.

Example:
<input type="search">

19. submit
- Submits the form.

Example:
<input type="submit" value="Submit">

20. week
- Selects a week and year.

Example:
<input type="week">

21. url
- Accepts only valid website URLs.

Example:
<input type="url">

22. time
- Selects time only.
